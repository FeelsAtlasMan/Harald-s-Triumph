		806 = { add_core = MCD }
		819 = { add_core = MCD }
		805 = { add_core = MCD }
		832 = { add_core = MCD }
		4132 = { add_core = MCD }
		4134 = { add_core = MCD }
		4135 = { add_core = MCD }
		release = MCD
		add_casus_belli = {
	          target = MCD
	          type = conquest
	          months = 12
        }
		war = {
	          target = MCD
	          attacker_goal = { casus_belli = conquest }
	          defender_goal = { casus_belli = status_quo }
	          call_ally = no
        }
		MCD = {
		    capital_scope = {
                sub_unit = { type = irregular value = current }
                sub_unit = { type = irregular value = current }
                sub_unit = { type = irregular value = current }
                sub_unit = { type = artillery value = current }
            }
		}
		MCD = {
		    define_general = {
	            name = "Kliment Zlatev"
	            personality = resolute
	            background = priest
            }
		}
		set_country_flag = macedonian_peasent_revolt
	}
}

#Reports from Macedonia
country_event = {
	id = 1122113
	title = "Reports from Macedonia"
	desc = "EVTDESC1122113"
	fire_only_once = yes
	
	trigger = {
		tag = BYZ
		MCD = { exists = no }
		has_country_flag = macedonian_peasent_revolt
	}
	
	option = {
		name = "A deserved punishment!"
		any_pop = { limit = { location = { OR = { is_core = MCD owned_by = THIS } } } reduce_pop = 0.96 }
		any_owned = { limit = { is_core = MCD controlled_by = THIS } add_province_modifier = { name = highlander_raids duration = 36500 } }
		any_owned = { limit = { is_core = MCD controlled_by = THIS } add_province_modifier = { name = war_torn duration = 365 } }
		set_country_flag = defeated_macedonian_peasent_revolt
	}
}
